7-2 换硬币
分数 10
作者 C课程组
单位 浙江大学
将一笔零钱换成5分、2分和1分的硬币，要求每种硬币至少有一枚，有几种不同的换法？

输入格式:
输入在一行中给出待换的零钱数额x∈(8,100)。

输出格式:
要求按5分、2分和1分硬币的数量依次从大到小的顺序，输出各种换法。每行输出一种换法，格式为：“fen5:5分硬币数量, fen2:2分硬币数量, fen1:1分硬币数量, total:硬币总数量”。最后一行输出“count = 换法个数”。

13
输出样例:
fen5:2, fen2:1, fen1:1, total:4
fen5:1, fen2:3, fen1:2, total:6
fen5:1, fen2:2, fen1:4, total:7
fen5:1, fen2:1, fen1:6, total:8
count = 4
*/
#include<stdio.h>
int main()
{
    int N,count=0,total=0;
    int fen5,fen2,fen1;
    scanf("%d",&N); 
    for(fen5=N/5;fen5>0;fen5--)
    {
       for(fen2=(N-fen5*5)/2;fen2>0;fen2--)
       {
           for(fen1=(N-fen5*5-fen2*2);fen1>0;fen1--)
           {
           if(fen5*5+fen2*2+fen1==N)
           {
               count++;
               total=fen1+fen2+fen5;
               printf("fen5:%d, fen2:%d, fen1:%d, total:%d\n",fen5,fen2,fen1,total);
            }
            }
       }
 
    }printf("count = %d",count);
}

/*
7-4 打印菱形图案
分数 10
作者 C课程组
单位 浙江大学
本题要求编写程序，打印一个高度为n的、由“*”组成的正菱形图案。

输入格式:
输入在一行中给出一个正的奇数n。

输出格式:
输出由n行星号“*”组成的菱形，如样例所示。每个星号后跟一个空格。

输入样例:
7
输出样例:
      * 
    * * * 
  * * * * * 
* * * * * * * 
  * * * * * 
    * * * 
      * 
*/



#include <stdio.h>
#include <math.h>
int main()
{
	int i, j, n, k;
	printf("input odd number n:");
	scanf("%d", &n);
	k = (n + 1) / 2; // 计算菱形的边长
	// 上三角
	for (i = 1; i <= k; i++)
	{
		for (j = 1; j <= n - 2 * i + 1; j++) // 循环打印空格
		{
			printf(" ");
		}
		for (j = 1; j <= 2 * i - 1; j++)
		{
			printf("* "); // 循环打印*
		}
		printf("\n");
	}
	// 下三角
	for (i = 1; i < k; i++)
	{
		for (j = 1; j <= i * 2; j++) // 打印空格
		{
			printf(" ");
		}
		for (j = 1; j <= n - 2 * i; j++)
		{
			printf("* ");
		}
		printf("\n");
	}
	return 0;
}

输出格式：

123456

*/

#include<stdio.h>
int main()
{
    int n,d,t;
    int mask=1;
    scanf("%d",&n);
    t=n;
    while(t>9)
    {
        t/=10;
        mask*=10;   
    }
    while(mask!=0)
    {
        d=n/mask;
        printf("%d ",d);
        
        n%=mask;
        mask/=10;
    }
}

输入格式：
输入在一行中给出一个(2, 2 000 000 000]范围内的偶数N。

输出格式：
在一行中按照格式“N = p + q”输出N的素数分解，其中p ≤ q均为素数。又因为这样的分解不唯一（例如24还可以分解为7+17），要求必须输出所有解中p最小的解。

24
输出样例：
24 = 5 + 19
*/

#include<stdio.h>
int f(int n);
int main()
{
    int n,i;
    scanf("%d",&n);
    for(i=2;i<=n;i++)
    {
       if(f(i)==1&&f(n-i)==1) 
       {
           printf("%d = %d + %d",n,i,n-i);
           break;
       }
    }
}
int f(int n)
{
    int i,flag=1,result;
    for(i=2;i<n;i++)
    {
        if(n%i==0)
        {
            flag=0;
            break;
        }
    }
    if(flag==1)
        result=1;
    else
        result=0;
    return result;
}
